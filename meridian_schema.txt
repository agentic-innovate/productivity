    'engagement'

-- ============================================================================
-- Core Entities
-- ============================================================================

CREATE TABLE regions (
    id          VARCHAR(100)  NOT NULL PRIMARY KEY,
    name        VARCHAR(100)  NOT NULL UNIQUE,
    description TEXT
);

COMMENT ON TABLE regions IS 'Geographic regions grouping branches for regional management';

CREATE TABLE branches (
    id              VARCHAR(10)   NOT NULL PRIMARY KEY,
    name            VARCHAR(200)  NOT NULL UNIQUE,
    region_id       VARCHAR(100)  NOT NULL REFERENCES regions(id),
    advisor_count   INTEGER,
    client_count    INTEGER,
    avg_engagement  INTEGER       CHECK (avg_engagement BETWEEN 0 AND 100),
    capacity        INTEGER       CHECK (capacity BETWEEN 0 AND 100)
);

COMMENT ON TABLE branches IS 'Physical branch offices where advisors operate';

CREATE INDEX idx_branches_region ON branches(region_id);

CREATE TABLE advisors (
    id               VARCHAR(10)   NOT NULL PRIMARY KEY,
    name             VARCHAR(200)  NOT NULL,
    initials         VARCHAR(4),
    branch_id        VARCHAR(10)   NOT NULL REFERENCES branches(id),
    capacity         INTEGER       CHECK (capacity BETWEEN 0 AND 100),
    client_book_size INTEGER,
    aum              VARCHAR(30),
    retention_rate   INTEGER       CHECK (retention_rate BETWEEN 0 AND 100),
    nps_score        INTEGER       CHECK (nps_score BETWEEN 0 AND 100),
    credentials      VARCHAR(100)
);

COMMENT ON TABLE advisors IS 'Wealth advisors with specializations, capacity, and performance metrics';

CREATE INDEX idx_advisors_branch ON advisors(branch_id);
CREATE INDEX idx_advisors_capacity ON advisors(capacity);
CREATE INDEX idx_advisors_nps ON advisors(nps_score);

CREATE TABLE advisor_specializations (
    id              BIGSERIAL     NOT NULL PRIMARY KEY,
    advisor_id      VARCHAR(10)   NOT NULL REFERENCES advisors(id) ON DELETE CASCADE,
    specialization  VARCHAR(100)  NOT NULL,
    UNIQUE (advisor_id, specialization)
);

COMMENT ON TABLE advisor_specializations IS 'Many-to-many relationship between advisors and their specialization areas';

CREATE INDEX idx_adv_specs_specialization ON advisor_specializations(specialization);

CREATE TABLE clients (
    id                  VARCHAR(20)       NOT NULL PRIMARY KEY,
    name                VARCHAR(200)      NOT NULL,
    type                client_type       NOT NULL,
    tier                wealth_tier       NOT NULL,
    aum                 VARCHAR(30),
    branch_id           VARCHAR(10)       NOT NULL REFERENCES branches(id),
    status              assignment_status NOT NULL DEFAULT 'unassigned',
    assigned_advisor_id VARCHAR(10)       REFERENCES advisors(id) ON DELETE SET NULL,
    source              VARCHAR(300),
    acquiring_banker    VARCHAR(200),
    scenario_key        TEXT
);

COMMENT ON TABLE clients IS 'Wealth management clients (prospects and acquired) awaiting or having completed advisor assignment';

CREATE INDEX idx_clients_branch ON clients(branch_id);
CREATE INDEX idx_clients_tier ON clients(tier);
CREATE INDEX idx_clients_status ON clients(status);
CREATE INDEX idx_clients_assigned_advisor ON clients(assigned_advisor_id);

CREATE TABLE users (
    id          BIGSERIAL     NOT NULL PRIMARY KEY,
    name        VARCHAR(200)  NOT NULL,
    email       VARCHAR(200)  NOT NULL UNIQUE,
    employee_id VARCHAR(20),
    role        user_role     NOT NULL
);

COMMENT ON TABLE users IS 'Platform users who log in via SSO with role-based access (BM, RM, HoS)';


-- ============================================================================
-- Pipeline / Assignment Tables
-- ============================================================================

CREATE TABLE assignment_runs (
    id                  VARCHAR(50)       NOT NULL PRIMARY KEY,
    client_id           VARCHAR(20)       NOT NULL REFERENCES clients(id),
    advisor_id          VARCHAR(10)       REFERENCES advisors(id) ON DELETE SET NULL,
    compatibility_score INTEGER           CHECK (compatibility_score BETWEEN 0 AND 100),
    compliance_status   compliance_status,
    duration_ms         INTEGER,
    status              run_status        NOT NULL DEFAULT 'pending',
    run_type            VARCHAR(50),
    created_at          TIMESTAMP         NOT NULL DEFAULT NOW(),
    completed_at        TIMESTAMP
);

COMMENT ON TABLE assignment_runs IS 'Orchestrated AI pipeline runs for client-advisor matching';

CREATE INDEX idx_runs_client ON assignment_runs(client_id);
CREATE INDEX idx_runs_advisor ON assignment_runs(advisor_id);
CREATE INDEX idx_runs_status ON assignment_runs(status);

CREATE TABLE pipeline_agents (
    id             BIGSERIAL     NOT NULL PRIMARY KEY,
